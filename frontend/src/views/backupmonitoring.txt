const transactionHistories = ref([]);

const fetchTransactionHistory = async () => {
    try {
        const response = await axios.get(`/transactionHistory`, {
            params: {
                page: currentPage.value,
                limit: perPage.value,
                search: searchQuery.value,
                barangay: searchQueryBarangay.value,
                date_from: dateFrom.value, // Include dateFrom in request

            },

        console.log(transactionHistories.value);

        totalTransactions.value = response.data.totalTransactions;
        currentPage.value = response.data.currentPage;
        perPage.value = response.data.perPage;

        // Handle empty results when not on the first page
        if (transactionHistories.value.length === 0 && currentPage.value > 1) {
            currentPage.value--;
            fetchTransactionHistory(); // Fetch updated page data
        }
    } catch (error) {
        console.error('Error fetching transaction history:', error);
    }
};

//Pages
const currentPage = ref(0);
const perPage = ref(0); // Number of customers per page
const totalTransactions = ref(0); // To track total customers for pagination

const goToPreviousPage = () => {
    if (currentPage.value > 1) {
        currentPage.value--;
        fetchTransactionHistory();
    }
};

const goToNextPage = () => {
    const totalPages = Math.ceil(totalTransactions.value / perPage.value); // Calculate total pages

        currentPage.value++;
        fetchTransactionHistory(); // Fetch users for the updated page
    }
};

const showOptionModal = ref(false);

};

    const modal = document.querySelector(".modal-content"); // Class of your modal content div

        closeOptionModal();
    }
};

const selectedCustomerId = ref(null);

// Watch to update customer_id when a history ID is selected
watch(selectedHistoryId, () => {
    const selectedTransaction = previoustransactionHistories.value.find(
        (transaction) => transaction.transaction_id === selectedHistoryId.value
    );
    if (selectedTransaction) {
        selectedCustomerId.value = selectedTransaction.customer_id;
    } else {
        selectedCustomerId.value = null;
    }
});


const sortedTransactions = computed(() => {
    // Filter transactions based on matching conditions
    let filteredTransactions = previoustransactionHistories.value.filter((transaction) => {
        // Include transactions matching either selectedCustomerId or selectedHistoryId
        const matchesCustomer = selectedCustomerId.value
            ? transaction.customer_id === selectedCustomerId.value
            : true;
        const matchesHistory = selectedHistoryId.value
            ? transaction.transaction_id === selectedHistoryId.value
            : true;

        return matchesCustomer || matchesHistory;
    });

    // Sort the filtered transactions
    return [...filteredTransactions].sort((a, b) => {
        // Primary sort: By date (most recent first)
        const dateDiff = new Date(b.transaction_date) - new Date(a.transaction_date);
        if (dateDiff !== 0) return dateDiff;

        // Secondary sort: By ID (descending)
        return b.transaction_history_id - a.transaction_history_id;
    });
});
